པ་རི་ཤུ་དྡྷི་མ་ན་སེ། ཨ་བྷྱ་རྩི་ཏ་པྲ་ཏིཥྵི་ཏཱ་ནཱཾ། ན་མོ་བྷ་ག་ཝ་ཏེ། ཨ་མྲྀ་ཏཱ་ཡུ་རྵ་སྱ། ཏ་ཐཱ་ག་ཏ་སྱ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཤུ་དྡྷ། ཨཱ་ཡུར་བི་ཤོ་དྷ་ནི། སཾ་ཧ་ར་སཾ་ཧ་ར། སརྦ་ཏ་ཐཱ་ག་ཏ་བཱིརྻ་བ་ལེ་ན་པྲ་ཏི་སཾ་ཧ་ར་ཨཱ་ཡུཿ ས་ར་ས་ར། སརྦ་ཏ་ཐཱ་ག་ཏ། ས་མ་ཡ། བོ་དྷི་བོ་དྷི། བུདྡྷྱ་བུདྡྷྱ། བོ་དྷ་ཡ་བོ་དྷ་ཡ། མ་མ་སརྦ་པཱ་པཾ་ཨཱ་ཝ་ར་ཎ་བི་ཤུ་དྡྷེ། བི་ག་ཏ་མ་ལཾ། ཚ་ར་སུ་བུ་དྡྷ་བུ་དྡྷེ་ཧུ་རུ་ཧུ་རུ་སྭཱ་ཧཱ། རྩ་སྔགས་སོ། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་མ་ལ་བི་ཤོ་དྷ་ནི། རུ་དྷ་བ་ལེ་པྲ་ཏི་སཾ་སྐཱ་ར། ཏ་ཐཱ་ག་ཏ་དྷཱ་ཏུ་དྷ་རེ། དྷ་ར་དྷ་ར། སཾ་དྷ་ར་སཾ་དྷ་ར། སརྦ་ཏ་ཐཱ་ག་ཏ་ཨ་དྷི་ཥྛཱ་ན་ཨ་དྷི་ཥྛཱི་ཏེ་སྭཱ་ཧཱ། སྙིང་པོའོ། །དེ་དག་ནངས་རེ་ཚར་རེ་མ་ཆགས་པར་བཟླ་ཞིང་ལན་མང་དུ་གྲུབ་ནས་མཆོག་གོ །དེའང་གཟུངས་མདོ་ཉིད་ལས། གང་ལ་ལ་ཞིག་གིས་གཙང་སྦྲ་ཆེར་བྱས་ཏེ་ཉིན་གཅིག་བཞིན་དུ་རྟག་པར་གསང་སྔགས་ལན་རེ་རེ་ཇི་སྲིད་འཚོའི་བར་དུ་བརྟེན་ན། ་་་་་་་་་་་་་་ལོངས་སྤྱོད་དང་བདེ་བ་མི་ཟད་པའི་གནས་སུ་སྐྱེ་བར་འགྱུར་རོ། །རྟག་པར་དུས་ཐམས་ཅད་དུ་བཟླས་བརྗོད་བྱས་ན་སངས་རྒྱས་བཞུགས་པའི་ཞིང་དུ་སྐྱེ་བར་འགྱུར་ཞིང་ཚེ་རབས་རྗེས་སུ་དྲན་པར་ཡང་འགྱུར་རོ། །དུས་མ་ཡིན་པའི་འཆི་བ་དང་སྒྲིབ་པ་ཐམས་ཅད་བྱང་བར་འགྱུར་རོ། །ཞེས་པས་མཚོན་ཏེ་ཕན་ཡོན་རྒྱ་ཆེར་གསུངས་པ་ལ་དད་པས་ཡི་དམ་དུ་བྱའོ། །ཡང་གཙུག་ཏོར་དྲི་མེད་ཀྱི་གཟུངས་སོགས་བཟླ་བར་སྤྲོ་ནའང་རྣལ་འབྱོར་འདི་ཁོ་ནས་བཀླག་པར་བྱ་སྟེ། གཟུངས་འདིའི་འཕྲོས་སམ། འདིའི་ཚབ་ཏུ་བཟླ་བ་གང་འདོད་བྱ་ཆོག་གོ །མཐར་སངས་རྒྱས་འཁོར་བཅས་མི་དམིགས་པར་བཞག བདག་གི་དགེ་བའི་ལས་རྣམས་འདི་དག་གིས། །འཇིག་རྟེན་འདིར་ནི་མྱུར་དུ་འཚང་རྒྱ་ཤོག །འགྲོ་ལ་ཕན་ཕྱིར་ཆོས་རྣམས་སྟོན་བགྱིད་ཅིང༌། །སེམས་ཅན་སྡུག་བསྔལ་མང་པོས་གཟིར་ལས་སྒྲོལ། །ཞེས་སོགས་བསྔོ་སྨོན་གྱི་རྒྱས་གདབ་བོ། །དེང་སང་རྩེ་གཅིག་སྒོམ་ནུས་དཀོན།

ཀློག་པའི་ཁ་ཁྲལ་བརྩོན་པ་མང༌། རྒྱལ་བས་བཀླག་པར་གདམས་འདི་ལ། ཀློག་པའི་སྐལ་པ་ལྡན་ན་རྨད། །ཅེས་པ་འདིའང་ཆོས་རྗེ་ཐམས་ཅད་མཁྱེན་པ་བུ་སྟོན་ཡབ་སྲས་དང༌། ངོར་ཆེན་རྡོ་རྗེའི་འཆང་ལས་བརྒྱུད་པའི་གདམས་ངག་ལག་ལེན་དུ་ཡོན་ཏན་རྒྱ་མཚོས་བསྡེབས་པ་གཞན་ལ་ཕན་པར་གྱུར་ཅིག། །།སརྦ་མངྒལཾ།། །།
