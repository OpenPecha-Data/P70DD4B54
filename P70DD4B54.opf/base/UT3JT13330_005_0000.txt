གཙུག་ཏོར་དྲི་མེད་གཟུངས་རིང་ནི། ན་མཿསརྦ་ཏཐཱ་ག་ཏཱ་ནཱཾ། ཨོཾ་མ་ཧཱ་ཙིནྟཱ་མ་ཎི་ཛྭ་ལ་ན་སཱ་ག་ར་གམྦྷཱིརེ། ཨཱ་ཀརྵ་ཡ། ཨཱ་ཀརྵ་ཡ། ཨཱ་ཡུནྡྷར། ཨཱ་ཡུནྡྷར། སནྡྷ་ར་སནྡྷ་ར། ཀྵཎ་ཀྵཎ། ཀྵེ་ཎི་ཀྵེ་ཎི། ཀྵུ་ཎུ་ཀྵུ་ཎུ། སརྦ་ཏ་ཐཱ་ག་ཏ་མ་ཧཱ་ས་མ་ཡ། ཏིཥྛ་ཏིཥྛ། མ་ཧཱ་བྷུ་བ་ན། སཱ་ག་ར། སཾ་ཤོ་དྷ་ཡ། མཱཾ་སརྦ་སྭ་ཏྭཱཾ་ཤྩ། བྷ་ག་བ་ཏི། སརྦ་པཱ་པྃ་བི་མ་ལེ། ཛ་ཡ་ཛ་ཡ། ཛ་ཡ་ལ་བྡྷེ། སྥུཊ་སྥུཊ། སྥོ་ཊ་ཡ་སྥོ་ཊ་ཡ། བི་ག་ཏེ་ཨཱ་བ་ར་ཎི། བྷ་ཡ་ཧ་ར་ཎི། ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། མྲྀ་ཏྱུ་དཎྜ་བྷ་རེ། ཨ་བྷ་ཡ་པྲ་དེ། ཨུཥྞཱི་ཥ་བྱཱ་ལོ་ཀི་ཏེ། ས་མནྟ་མུ་ཥེ། ས་མནྟ་བྱཱ་བ་ལོ་ཀི་ཏེ། མ་ཧཱ་མཱ་ཡེ། མ་ཧཱ་བཱ་ཤ་དྷ་རེ། ཨ་མོ་གྷ་བི་མ་ལེ། ཨཱ་ཀརྵ་ཡ། ཨཱ་ཀརྵ་ཡ། ཨཱ་ཀཌྜྷྱ་ཡ། ཨཱ་ཀཌྜྷྱ་ཡ། བྷ་ར་བྷ་ར། སམྦྷར། སམྦྷར། ཨིནྡྲ་ཡ་བི་ཤོ་དྷ་ནི་བྷཱུ་ཥི་ཏ་བྷུ་ཛེ། མ་ཧཱ་མུ་དྲཱ་བི་ལོ་ཀི་ཏེ། ཛ་ཡ་ཛ་ཡ། སིདྡྷེ་སིདྡྷེ། བུདྡྷེ་བུདྡྷེ། བོ་དྷ་ནི་བོ་དྷ་ནི། སཾ་བོ་དྷ་ནི་སཾ་བོ་དྷ་ནི། ཤོ་དྷ་ནི་ཤོ་དྷ་ནི། སཾ་ཤོ་དྷ་ནི་སཾ་ཤོ་དྷ་ནི། བི་ཤོ་དྷ་ནི་བི་ཤོ་དྷ་ནི། ཧ་ར་ཧ་ར། མ་མ་སརྦ་པཱ་པཾ། སརྦ་ཏ་ཐཱ་ག་ཏ་ཀུ་ལ་བྷུ་ཛེ། ས་མ་ཡ་ཏིཥྛེ། པྲ་ས་ར་ཏུ་མ་མ་པུཎྱེ། བི་ན་ཤྱནྟུ་པཱ་པཾ། སརྦ་ཀིལྦིཥ་ཧ་རེ། མ་ཎི་བི་ཤུདྡྷེ། ཤོ་དྷ་ཡ་བི་མ་ལེ། བི་ཀཱ་སི་ཏ་པདྨེ། ཀལྤ་ཙི་ཏེ་བྷུ་ཛེ། ཥཊྤཱ་ར་མི་ཏཱ་པ་རི་པཱུ་ར་ཎི། སརྦ་ཏ་ཐཱ་ག་ཏོ་ཥྞཱི་ཥ་བི་མ་ལེ་བི་ལོ་ཀི་ཏེ་སྭཱ་ཧཱ། སརྦ་ཏ་ཐཱ་ག་ཏ་གུ་ཧྱ་ཨདྷིཥྛཱ་ན་ཨདྷིཥྛི་ཏེ་སྭཱ་ཧཱ། ཨཱ་ཡུརྡ་དེ་སྭཱ་ཧཱ། པུཎྱ་ད་དེ་སྭཱ་ཧཱ། པུ་ཎྱ་བི་ལོ་ཀི་ཏེ་སྭཱ་ཧཱ། པུཎྱ་ཨ་བ་ལོ་ཀི་ཏེ་སྭཱ་ཧཱ། མྲྀ་ཏྱུ་ད་ཎྜེ་སྭཱ་ཧཱ། ཡ་མ་དཎྜེ་སྭཱ་ཧཱ། ཡ་མ་དཱ་ཏེ་སྭཱ་ཧཱ། སཾ་ཧ་ར་ཎི་སྭཱ་ཧཱ། སམྦཱ་ར་ཎི་སྭཱ་ཧཱ། སནྡྷ་ར་ཎི་སྭཱ་ཧཱ། པྲ་ཏི་ས་ར་ཎི་སྭཱ་ཧཱ། ཨོ་ཛོ་བ་ཏི་སྭཱ་ཧཱ། ཏེ་ཛོ་བ་ཏི་སྭཱ་ཧཱ། ཛ་ཡ་བ་ཏི་སྭཱ་ཧཱ། སརྦ་ཏ་ཐཱ་ག་ཏེ་མུ་དྲཱ་ཨ་དྷིཥྛཱ་ན་ཨ་དྷི་ཥྛི་ཏེ་སྭཱ་ཧཱ།

སྙིང་པོ་ནི། ཨོཾ་ན་མོ་སྟྲཻ་ཡ་དྷི་ཀཱ་ནཱཾ། སརྦ་ཏ་ཐཱ་ག་ཏ་ཧྲྀ་ད་ཡ་གརྦྷེ་ཛྭ་ལ་ཛྭ་ལ། དྷརྨ་དྷཱ་ཏུ་གརྦྷེ་སཾ་བྷ་ར་མ་མ་ཨཱ་ཡུཿ། སཾ་ཤོ་དྷ་ཡ། སཾ་ཤོ་དྷ་ཡ། མ་མ་སརྦ་པཱ་པཾ། སརྦ་ཏ་ཐཱ་ག་ཏ་ས་མནྟོ་ཥྞཱི་ཥ་བི་མ་ལེ་བི་ཤུདྡྷེ། ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། ཨྃ་བྃ་སྃ་ཛཿསྭཱ་ཧཱ། །
